create or replace function  calculate_salary (p_date date,p_sal number,p_empno number) return number
is

v_totalsal number(20);
v_empno number(20);

begin
select empno,((sal*(months_between(trunc(sysdate),p_date)))/(trunc(sysdate) - p_date))*(trunc(sysdate) - p_date)
into v_totalsal,v_empno
from scott.emp
where empno = p_empno;

return v_totalsal;
end;
